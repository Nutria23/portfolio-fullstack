import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import '../models/workout_model.dart';
import '../services/firebase_service.dart';
import '../widgets/progress_ring.dart';

class HomeScreen extends StatefulWidget {
  @override
  _HomeScreenState createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen> {
  late Future<DailyStats> _stats;

  @override
  void initState() {
    super.initState();
    _stats = FirebaseService.getDailyStats();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Color(0xFF0A0A1E),
      body: FutureBuilder<DailyStats>(
        future: _stats,
        builder: (ctx, snapshot) {
          if (!snapshot.hasData) return LoadingSpinner();
          final stats = snapshot.data!;
          return Column(children: [
            ProgressRing(progress: stats.goalPercent),
            StatsRow(
              calories: stats.calories,
              steps: stats.steps,
              activeMinutes: stats.activeMinutes,
            ),
            WorkoutList(workouts: stats.workouts),
          ]);
        },
      ),
    );
  }
}
