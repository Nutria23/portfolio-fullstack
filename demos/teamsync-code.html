// KanbanBoard.vue - Tablero Kanban principal
<template>
  <div class="kanban-board">
    <KanbanColumn
      v-for="column in columns"
      :key="column.id"
      :column="column"
      @task-moved="handleTaskMove"
      @task-created="handleTaskCreate"
    />
  </div>
</template>

<script setup lang="ts">
import { useQuery, useMutation } from '@vue/apollo-composable';
import { GET_BOARD, MOVE_TASK } from '@/graphql/queries';

const { result, loading } = useQuery(GET_BOARD);
const { mutate: moveTask } = useMutation(MOVE_TASK);

const columns = computed(() => result.value?.board?.columns ?? []);

async function handleTaskMove(taskId: string, toColumn: string) {
  await moveTask({ taskId, columnId: toColumn });
  // Actualización optimista ya manejada por Apollo
}

async function handleTaskCreate(columnId: string, title: string) {
  await createTask({ columnId, title, assignee: currentUser.id });
}
</script>
