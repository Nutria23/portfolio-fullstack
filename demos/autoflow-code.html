# autoflow.py - Sistema de automatización IoT
import asyncio
from dataclasses import dataclass
from typing import List, Dict
from mqtt_handler import MQTTClient
from aws_iot import IoTCore
from alerts import AlertManager

@dataclass
class SensorReading:
    device_id: str
    temperature: float
    humidity: float
    pressure: float
    timestamp: datetime

class AutoFlowEngine:
    def __init__(self, config: Dict):
        self.mqtt = MQTTClient(config['broker'])
        self.iot = IoTCore(region=config['aws_region'])
        self.alerts = AlertManager()
        self.pipelines: List[Pipeline] = []

    async def process_reading(self, reading: SensorReading):
        """Procesa lecturas de sensores IoT en tiempo real"""
        if reading.temperature > 28.0:
            await self.alerts.send(
                level='warning',
                message=f'Temp alta: {reading.temperature}°C',
                device=reading.device_id
            )

        await self.iot.publish(
            topic=f'sensors/{reading.device_id}/data',
            payload=reading.to_dict()
        )

    async def run(self):
        """Ejecuta el engine de automatización"""
        await self.mqtt.connect()
        async for reading in self.mqtt.subscribe('sensors/+/raw'):
            await self.process_reading(reading)

if __name__ == '__main__':
    engine = AutoFlowEngine(load_config())
    asyncio.run(engine.run())
