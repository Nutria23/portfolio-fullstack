import pandas as pd
from celery import Celery, chain
from sqlalchemy import create_engine
from report_generator import PDFReport
from notifications import send_slack, send_email

app = Celery('pipeline', broker='amqp://localhost')
engine = create_engine('postgresql://localhost/analytics')

@app.task
def transform_data(raw_data: dict) -> dict:
    """Limpia y transforma datos con Pandas"""
    df = pd.DataFrame(raw_data['records'])
    df = df.drop_duplicates()
    df['date'] = pd.to_datetime(df['date'])
    df['revenue'] = df['revenue'].astype(float)
    return df.to_dict(orient='records')

@app.task
def load_and_report(clean_data: list):
    """Carga datos y genera reportes"""
    df = pd.DataFrame(clean_data)
    df.to_sql('sales', engine, if_exists='append')
    
    report = PDFReport(df)
    path = report.generate()
    send_email('admin@empresa.com', attachment=path)
    send_slack('#data-team', f'📊 Reporte: {len(df)} registros')
