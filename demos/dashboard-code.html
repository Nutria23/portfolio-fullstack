// Dashboard.jsx - Componente principal de analíticas
import React, { useState, useEffect } from 'react';
import { BarChart, LineChart, PieChart } from 'recharts';
import { fetchAnalytics, generateReport } from '../api/analytics';
import KPICard from '../components/KPICard';

const Dashboard = () => {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [dateRange, setDateRange] = useState('monthly');

  useEffect(() => {
    const loadData = async () => {
      try {
        const analytics = await fetchAnalytics(dateRange);
        setData(analytics);
      } catch (err) {
        console.error('Error loading analytics:', err);
      } finally {
        setLoading(false);
      }
    };
    loadData();
  }, [dateRange]);

  const kpis = [
    { title: 'Ingresos', value: data?.revenue, icon: '💰', change: +12.5 },
    { title: 'Usuarios', value: data?.users, icon: '👥', change: +8.3 },
    { title: 'Conversión', value: data?.conversion, icon: '📈', change: +0.8 },
    { title: 'Tickets', value: data?.tickets, icon: '🎫', change: -15 },
  ];

  return (
    <div className="dashboard-container">
      <header className="dashboard-header">
        <h1>Dashboard Analytics</h1>
        <DateRangePicker value={dateRange} onChange={setDateRange} />
      </header>

      <div className="kpi-grid">
        {kpis.map((kpi, i) => (
          <KPICard key={i} {...kpi} loading={loading} />
        ))}
      </div>

      <div className="charts-row">
        <BarChart data={data?.salesByMonth} />
        <PieChart data={data?.trafficSources} />
      </div>
    </div>
  );
};

export default Dashboard;
